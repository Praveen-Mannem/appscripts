/**
 * Google Apps Script to list inactive users and their license types.
 *
 * INSTRUCTIONS:
 * 1. Create a new Google Sheet.
 * 2. Go to Extensions > Apps Script.
 * 3. Paste this code.
 * 4. Add Services (left sidebar):
 *    - AdminDirectory
 *    - AdminLicenseManager
 * 5. Run the 'exportInactiveUsers' function.
 */

const CONFIG = {
  // 'my_customer' is the special alias for the current Google Workspace account
  customerId: 'my_customer', 
  inactivityDays: 365,
  // Product ID for Google Workspace (usually 'Google-Apps')
  productId: 'Google-Apps' 
};

// Map of common Google Workspace SKU IDs to Names
// You can find more SKUs here: https://developers.google.com/admin-sdk/licensing/v1/how-tos/products
const SKU_MAP = {
  '1010020020': 'Google Workspace Enterprise Plus',
  '1010020026': 'Google Workspace Enterprise Standard',
  '1010020027': 'Google Workspace Business Starter',
  '1010020028': 'Google Workspace Business Standard',
  '1010020025': 'Google Workspace Business Plus',
  '1010060001': 'Google Workspace Essentials',
  '1010060003': 'Google Workspace Enterprise Essentials',
  '1010020029': 'Google Workspace Enterprise Essentials Plus',
  '1010020010': 'Google Workspace Frontline',
  '1010010001': 'G Suite Basic',
  '1010050001': 'G Suite Business',
  '1010310002': 'Cloud Identity Free',
  '1010310003': 'Cloud Identity Premium'
};

function exportInactiveUsers() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetName = 'Inactive Users Report';
  let sheet = ss.getSheetByName(sheetName);
  
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
  } else {
    sheet.clear();
  }
  
  // Set Headers
  sheet.appendRow([
    'Email', 
    'Full Name', 
    'Last Login Time', 
    'Status', 
    'Days Inactive', 
    'License Name', 
    'License SKU ID'
  ]);
  
  // Format header row
  sheet.getRange(1, 1, 1, 7).setFontWeight('bold').setBackground('#f3f3f3');
  
  const cutoffDate = new Date();
  cutoffDate.setDate(cutoffDate.getDate() - CONFIG.inactivityDays);
  
  Logger.log(`Starting search for users inactive since ${cutoffDate.toISOString()}...`);
  
  let pageToken;
  let totalProcessed = 0;
  let inactiveCount = 0;
  
  do {
    // List users
    const response = AdminDirectory.Users.list({
      customer: CONFIG.customerId,
      maxResults: 200,
      pageToken: pageToken,
      orderBy: 'email',
      projection: 'full' // Needed to get lastLoginTime
    });
    
    const users = response.users;
    
    if (users && users.length > 0) {
      const rows = [];
      
      for (const user of users) {
        totalProcessed++;
        
        const lastLoginStr = user.lastLoginTime;
        let lastLoginDate = new Date(0); // Default to epoch if never logged in
        let status = 'NEVER_LOGGED_IN';
        
        if (lastLoginStr && lastLoginStr !== '0') {
          lastLoginDate = new Date(lastLoginStr);
          status = 'ACTIVE'; // Tentative, will check date
        }
        
        // Check if suspended
        if (user.suspended) {
          status = 'SUSPENDED';
        }
        
        // Check inactivity
        // If never logged in, they are considered inactive
        // If logged in, check if before cutoff
        const isInactive = (lastLoginStr === '0' || !lastLoginStr) || (lastLoginDate < cutoffDate);
        
        if (isInactive) {
          inactiveCount++;
          if (status === 'ACTIVE') status = 'INACTIVE'; // Update status for reporting
          
          // Calculate days inactive
          const daysInactive = lastLoginStr && lastLoginStr !== '0' 
            ? Math.floor((new Date() - lastLoginDate) / (1000 * 60 * 60 * 24)) 
            : 'Never';

          // Fetch License
          const licenseInfo = getUserLicense(user.primaryEmail);
          
          rows.push([
            user.primaryEmail,
            user.name ? user.name.fullName : '',
            lastLoginStr === '0' ? 'Never' : lastLoginDate.toISOString(),
            status,
            daysInactive,
            licenseInfo.name,
            licenseInfo.skuId
          ]);
        }
      }
      
      // Write batch to sheet
      if (rows.length > 0) {
        sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, rows[0].length).setValues(rows);
      }
    }
    
    pageToken = response.nextPageToken;
  } while (pageToken);
  
  Logger.log(`Finished. Processed ${totalProcessed} users. Found ${inactiveCount} inactive users.`);
  
  if (inactiveCount === 0) {
    Browser.msgBox('No inactive users found.');
  } else {
    Browser.msgBox(`Process Complete. Found ${inactiveCount} inactive users.`);
  }
}

/**
 * Helper to get the license for a specific user.
 * Uses AdminLicenseManager service.
 */
function getUserLicense(userEmail) {
  try {
    const response = AdminLicenseManager.LicenseAssignments.listForProduct(
      CONFIG.productId, 
      CONFIG.customerId, 
      {
        userId: userEmail
      }
    );
    
    if (response.items && response.items.length > 0) {
      // A user might have multiple assignments, but usually one main one for Google-Apps.
      // We'll take the first one.
      const skuId = response.items[0].skuId;
      const name = SKU_MAP[skuId] || 'Unknown SKU (' + skuId + ')';
      return { skuId: skuId, name: name };
    }
    
    return { skuId: 'None', name: 'No License Assigned' };
    
  } catch (e) {
    Logger.log(`Error fetching license for ${userEmail}: ${e.message}`);
    return { skuId: 'Error', name: 'Error: ' + e.message };
  }
}
